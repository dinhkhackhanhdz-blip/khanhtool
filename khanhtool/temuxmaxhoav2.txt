#!/bin/bash

VERSION="4.68"
BASE_URL="https://raw.githubusercontent.com/khanhitxiaomi/khanhitxiaomi/main/khanhxiaomi"

set +x 2>/dev/null

RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
WHITE='\033[1;37m'
NC='\033[0m'

# ==============================
# LOADING
# ==============================
loading() {
  echo -ne "${CYAN}⏳ Đang xử lý"
  for i in {1..3}; do
    echo -ne "."
    sleep 0.4
  done
  echo -e "${NC}"
}

# ==============================
# CHECK KEY
# ==============================
check_key() {
  DEVICE=$(getprop ro.serialno 2>/dev/null || hostname)

  while true; do
    clear

    echo ""
    echo -e "${CYAN}📺 CÔNG CỤ CÀI ĐẶT TIVI XIAOMI BẢN QUYỀN${NC}"
    echo -e "${YELLOW}⚡ TỰ ĐỘNG - NHANH - ỔN ĐỊNH - UPDATE${NC}"
    echo -e "${GREEN}👨‍💻 Khánh IT | 0967.341.608 | ${VERSION}${NC}"
    echo ""

    echo -e "${BLUE}🔑 VUI LÒNG NHẬP MẬT KHẨU${NC}"
    echo -e "${WHITE}➤ KEY phân biệt chữ HOA / chữ thường${NC}"
    echo ""
    read -p "👉 Nhập KEY: " USER_KEY
    USER_KEY=$(echo "$USER_KEY" | xargs)

    [ -z "$USER_KEY" ] && echo -e "${RED}❌ Chưa nhập KEY${NC}" && sleep 1.5 && continue

    loading

    DATA=$(curl -fsSL --connect-timeout 5 "$BASE_URL/keys.txt")

    [ -z "$DATA" ] && echo -e "${RED}❌ Lỗi server${NC}" && sleep 2 && continue

    LINE=$(echo "$DATA" | grep -m 1 "^${USER_KEY}|")

    [ -z "$LINE" ] && echo -e "${RED}❌ KEY sai${NC}" && sleep 1.5 && continue

    STATUS=$(echo "$LINE" | cut -d'|' -f2)
    KEY_DEVICE=$(echo "$LINE" | cut -d'|' -f3)

    [ "$STATUS" != "active" ] && echo -e "${RED}❌ KEY bị khóa${NC}" && sleep 1.5 && continue

    if [ -n "$KEY_DEVICE" ] && [ "$KEY_DEVICE" != "$DEVICE" ]; then
      echo -e "${RED}❌ Sai thiết bị${NC}"
      sleep 1.5
      continue
    fi

    echo -e "${GREEN}✅ XÁC THỰC THÀNH CÔNG${NC}"
    sleep 1
    break
  done
}

# ==============================
# UPDATE
# ==============================
check_update() {
  NEW_VERSION=$(curl -fsSL --connect-timeout 5 "$BASE_URL/version.txt")

  [ -z "$NEW_VERSION" ] && return

  if [ "$NEW_VERSION" != "$VERSION" ]; then
    echo -e "${YELLOW}🚀 ĐANG UPDATE...${NC}"
    loading
    curl -fsSL -o tool.sh "$BASE_URL/tool.sh"
    chmod +x tool.sh
    echo -e "${GREEN}✅ Update xong - chạy lại${NC}"
    exit 0
  fi
}

# ==============================
# LOAD CORE
# ==============================
load_core() {
  echo -e "${BLUE}📦 ĐANG LOAD CORE...${NC}"

  DIR="$HOME/storage/shared/apktool"
  TMP="$DIR/core.txt"

  mkdir -p "$DIR"

  loading

  if ! curl -fsSL --connect-timeout 10 "$BASE_URL/core.txt" -o "$TMP"; then
    echo -e "${RED}❌ Lỗi tải core${NC}"
    exit 1
  fi

  [ ! -s "$TMP" ] && echo -e "${RED}❌ File rỗng${NC}" && exit 1

  CODE=$(base64 -d "$TMP" 2>/dev/null)

  [ -z "$CODE" ] && echo -e "${RED}❌ Decode lỗi${NC}" && exit 1

  echo -e "${GREEN}🚀 KHỞI ĐỘNG TOOL...${NC}"
  sleep 1

  echo "$CODE" | bash
}

# ==============================
# RUN
# ==============================
clear

echo -e "${GREEN}🚀 TOOL TIVI XIAOMI${NC}"
echo -e "${CYAN}📺 Cài đặt & tối ưu tự động${NC}"
echo ""
echo -e "${YELLOW}⚡ ADB • APK • Android TV${NC}"
echo ""
echo -e "${GREEN}👨‍💻 Khánh IT | ${VERSION}${NC}"
echo ""
echo -e "${BLUE}⏳ Đang khởi động...${NC}"

check_key
check_update
load_core
