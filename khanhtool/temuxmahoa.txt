#!/bin/bash

VERSION="4.68"
BASE_URL="https://raw.githubusercontent.com/khanhitxiaomi/khanhitxiaomi/main/khanhxiaomi"

set +x 2>/dev/null

# ===== COLOR =====
RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
NC='\033[0m'

# ===== CHECK KEY =====
check_key() {
  echo -e "${CYAN}🔐 CHECK KEY...${NC}"

  # FIX INPUT TERMUX
  echo -ne "🔐 Nhập key: "
  read -r USER_KEY

  [ -z "$USER_KEY" ] && {
    echo -e "${RED}❌ KEY TRỐNG${NC}"
    exit 1
  }

  DEVICE=$(getprop ro.serialno 2>/dev/null || hostname)

  DATA=$(curl -s --connect-timeout 10 "$BASE_URL/keys.txt")

  [ -z "$DATA" ] && {
    echo -e "${RED}❌ KHÔNG LOAD ĐƯỢC KEY SERVER${NC}"
    exit 1
  }

  LINE=$(echo "$DATA" | grep "^$USER_KEY|")

  [ -z "$LINE" ] && {
    echo -e "${RED}❌ KEY SAI${NC}"
    exit 1
  }

  STATUS=$(echo "$LINE" | cut -d'|' -f2)
  KEY_DEVICE=$(echo "$LINE" | cut -d'|' -f3)

  [ "$STATUS" != "active" ] && {
    echo -e "${RED}❌ KEY BỊ KHÓA${NC}"
    exit 1
  }

  if [ -n "$KEY_DEVICE" ] && [ "$KEY_DEVICE" != "$DEVICE" ]; then
    echo -e "${RED}❌ SAI THIẾT BỊ${NC}"
    exit 1
  fi

  echo -e "${GREEN}✅ KEY OK${NC}"
}

# ===== UPDATE =====
check_update() {
  echo -e "${BLUE}🔄 CHECK UPDATE...${NC}"

  NEW=$(curl -s --connect-timeout 10 "$BASE_URL/version.txt")

  [ -z "$NEW" ] && {
    echo -e "${YELLOW}⚠️ KHÔNG CHECK ĐƯỢC VERSION${NC}"
    return
  }

  if [ "$NEW" != "$VERSION" ]; then
    echo -e "${YELLOW}🚀 UPDATE TOOL...${NC}"

    curl -s -o tool.sh "$BASE_URL/tool.sh"
    chmod +x tool.sh

    echo -e "${GREEN}✅ UPDATE DONE - RUN AGAIN${NC}"
    exit 0
  fi
}

# ===== LOAD CORE =====
load_core() {
  echo -e "${CYAN}📦 LOAD CORE...${NC}"

  TMP="/data/data/com.termux/files/home/core.txt"

  curl -s --connect-timeout 10 "$BASE_URL/core.txt" -o "$TMP"

  [ ! -s "$TMP" ] && {
    echo -e "${RED}❌ LOAD CORE FAIL${NC}"
    exit 1
  }

  CODE=$(base64 -d "$TMP" 2>/dev/null)

  [ -z "$CODE" ] && {
    echo -e "${RED}❌ DECODE FAIL${NC}"
    exit 1
  }

  bash -c "$CODE"
}

# ===== MAIN =====
clear

echo -e "${GREEN}=============================="
echo -e " TOOL XIAOMI TV PRO LOADER"
echo -e " VERSION: $VERSION"
echo -e "==============================${NC}"

check_key
check_update
load_core
